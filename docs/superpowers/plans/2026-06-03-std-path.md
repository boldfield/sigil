# std/path Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Ship `std/path.sigil` — a pure-string POSIX path module
(`join`/`split`/`basename`/`dirname`/`splitext`/`normalize`/`is_absolute`)
matching Python `posixpath` semantics — closing the §14.1 backlog item.

**Architecture:** A new real-source stdlib module. It is auto-embedded
into the compiler via `include_dir!("$CARGO_MANIFEST_DIR/../std")`
(`compiler/src/stdlib_embed.rs`) and auto-indexed by
`stdlib_index::build()` (walks the embedded corpus) — so **no allow-list
or registration edits are needed**; dropping `std/path.sigil` in and
rebuilding is sufficient. Every function is pure (`![]`), implemented
over existing primitives: bare intrinsics (`string_concat`,
`string_length`, `string_substring`, `string_starts_with`,
`string_ends_with`, `string_index_of`) plus `std.string`
(`string_split`, `string_replace`) and `std.list` (`List`/`Cons`/`Nil`,
`fold`, `reverse`). No new types, intrinsics, or language features.

**Tech Stack:** Sigil source (`std/*.sigil`); Rust e2e harness
(`compiler/tests/e2e.rs`, helper `compile_and_run(source, name) ->
(stdout, stderr, code)`); `scripts/smoke.sh`, `scripts/reproducibility.sh`.

**Spec:** `docs/superpowers/specs/2026-06-03-std-path-design.md` — the
semantics contract there is authoritative; the test cases below are
copied from it.

## Reference: primitives used (all confirmed present)

- **Bare intrinsics** (prelude — callable without import): `string_concat(a, b) -> String`, `string_length(s) -> Int` (byte length), `string_substring(s, start, end) -> String` (half-open `[start, end)`, byte indices), `string_starts_with(s, prefix) -> Bool`, `string_ends_with(s, suffix) -> Bool`, `string_index_of(s, sub) -> Int` (first index, `< 0` if absent).
- **`std.string`** (needs `import std.string` + `use`): `string_split(s, sep) -> List[String]` (standard split — leading/trailing separators yield empty components; no-separator input yields a 1-element list), `string_replace(s, find, repl) -> String`.
- **`std.list`** (needs `import std.list` + `use`): `List[A]`, `Cons`, `Nil`, `fold(xs, init, f) -> B` where `f: (B, A) -> B`, `reverse(xs) -> List[A]`.
- Tuples are built-in: `(a, b)` literal, `-> (A, B)` return, `match t { (a, b) => ... }`.

```
//
// std/path.sigil — pure-string POSIX path manipulation.

// Slash-separated, no filesystem access (that is std.fs's job). Every

//   path_is_absolute("/")  = true   path_is_absolute("")  = false
fn path_is_absolute(p: String) -> Bool ![] {
```

```
// `path_join(a, b)` joins two path segments with "/", matching
// posixpath.join's binary case:
//   - if `b` is absolute, the result is `b` (absolute RESETS — a wart
//     inherited deliberately for prediction-match);
//   - else if `a` is empty, the result is `b`;
//   - else if `a` already ends with "/", concatenate directly;
//   - else insert a single "/".
//   path_join("a","b")="a/b"  path_join("a/","b")="a/b"
//   path_join("a","/b")="/b"  path_join("","b")="b"  path_join("a","")="a/"
fn path_join(a: String, b: String) -> String ![] {
  if path_is_absolute(b) {
    b
  } else {
    if string_length(a) == 0 {
      b
    } else {
      if string_ends_with(a, "/") {
        string_concat(a, b)
      } else {
        string_concat(string_concat(a, "/"), b)
      }
    }
  }
}
```

```
// --- shared helpers -----------------------------------------------------

// fold combinator returning the second arg — folds a list to its LAST
// element (or the seed for an empty list).
fn __take_second(acc: String, x: String) -> String ![] { x }

// Last element of a non-empty component list ("" for an empty list).
fn __last_component(comps: List[String]) -> String ![] {
  fold(comps, "", __take_second)
}

// True iff `s` is non-empty and consists only of "/" characters.
fn __is_all_slashes(s: String) -> Bool ![] {
  if string_length(s) == 0 {
    false
  } else {
    string_length(string_replace(s, "/", "")) == 0
  }
}

// Remove ALL trailing "/" from `s`. Caller guarantees `s` is not
// all-slashes, so this never empties a slash-only string.
fn __rstrip_slashes(s: String) -> String ![] {
  if string_ends_with(s, "/") {
    __rstrip_slashes(string_substring(s, 0, string_length(s) - 1))
  } else {
    s
  }
}

// posixpath's head normalization: strip trailing slashes from a split
// head UNLESS the head is empty or all-slashes (then keep it verbatim).
fn __strip_head_slashes(head: String) -> String ![] {
  if string_length(head) == 0 {
    head
  } else {
    if __is_all_slashes(head) {
      head
    } else {
      __rstrip_slashes(head)
    }
  }
}

// --- split / basename / dirname ----------------------------------------

// `path_split(p)` returns `(head, tail)` where `tail` is the final
// component (after the last "/") and `head` is everything before it
// (trailing slashes stripped unless head is all-slashes). Mirrors
// posixpath.split.
//   path_split("a/b/c")=("a/b","c")  path_split("a/b/")=("a/b","")
//   path_split("/a")=("/","a")  path_split("/")=("/","")  path_split("a")=("","a")
fn path_split(p: String) -> (String, String) ![] {
  if string_index_of(p, "/") < 0 {
    ("", p)
  } else {
    let tail: String = __last_component(string_split(p, "/"));
    // Index just past the last "/" = len(p) - len(tail).
    let cut: Int = string_length(p) - string_length(tail);
    let head_raw: String = string_substring(p, 0, cut);
    (__strip_head_slashes(head_raw), tail)
  }
}

// `path_basename(p)` is the final component (= path_split's tail).
//   path_basename("/a/b")="b"  path_basename("/a/b/")=""  path_basename("/")=""
fn path_basename(p: String) -> String ![] {
  match path_split(p) {
    (_, tail) => tail,
  }
}

// `path_dirname(p)` is everything before the final component (= head).
//   path_dirname("/a/b")="/a"  path_dirname("a")=""  path_dirname("/")="/"
fn path_dirname(p: String) -> String ![] {
  match path_split(p) {
    (head, _) => head,
  }
}
```

```
// True iff any element of `xs` is a non-empty string.
fn __or_nonempty(acc: Bool, x: String) -> Bool ![] {
  if acc { true } else { string_length(x) > 0 }
}
fn __any_nonempty(xs: List[String]) -> Bool ![] {
  fold(xs, false, __or_nonempty)
}

// Given a basename split on ".", does it have a real "stem" before the
// final dot? (i.e. some component other than the last is non-empty).
// Leading-dot dotfiles like ".bashrc" / "..a" have no real stem.
fn __has_real_stem(parts: List[String]) -> Bool ![] {
  match reverse(parts) {
    Nil => false,
    Cons(_, rest) => __any_nonempty(rest),
  }
}

// Extension of a basename (incl. the leading dot), or "" if none.
//   __ext_of_basename("a.tar.gz")=".gz"  __ext_of_basename(".bashrc")=""
//   __ext_of_basename("a.")="."  __ext_of_basename("a")=""
fn __ext_of_basename(b: String) -> String ![] {
  let parts: List[String] = string_split(b, ".");
  if __has_real_stem(parts) {
    string_concat(".", __last_component(parts))
  } else {
    ""
  }
}

// `path_splitext(p)` splits the trailing extension off `p`'s basename.
// `ext` includes the dot (or is ""); `root ++ ext == p` always.
//   path_splitext("a.tar.gz")=("a.tar",".gz")  path_splitext(".bashrc")=(".bashrc","")
//   path_splitext("a.")=("a",".")  path_splitext("/a/b.c")=("/a/b",".c")
fn path_splitext(p: String) -> (String, String) ![] {
  let ext: String = __ext_of_basename(path_basename(p));
  let root: String = string_substring(p, 0, string_length(p) - string_length(ext));
  (root, ext)
}
```

- [ ] **Step 3: Implement normalize + helpers**

Append to `std/path.sigil`. Note: component equality to `"."` / `".."`
is done via length + `string_starts_with` (avoids needing a string-eq
intrinsic): a length-1 string starting with `"."` IS `"."`; a length-2
string starting with `".."` IS `".."`.

```
// Component classifiers (exact-match via length + prefix).
fn __is_dot(s: String) -> Bool ![] {
  if string_length(s) == 1 { string_starts_with(s, ".") } else { false }
}
fn __is_dotdot(s: String) -> Bool ![] {
  if string_length(s) == 2 { string_starts_with(s, "..") } else { false }
}
// "" (from "//" or a leading/trailing slash) or "." — dropped in normpath.
fn __is_empty_or_dot(s: String) -> Bool ![] {
  if string_length(s) == 0 { true } else { __is_dot(s) }
}

// Push one component onto the reverse-ordered stack (most-recent first),
// applying normpath rules.
fn __norm_push(stack: List[String], comp: String, is_abs: Bool) -> List[String] ![] {
  if __is_empty_or_dot(comp) {
    stack
  } else {
    if __is_dotdot(comp) {
      match stack {
        Cons(top, rest) =>
          if __is_dotdot(top) {
            Cons(comp, stack)   // can't cancel a ".." — push another
          } else {
            rest                // pop the preceding real component
          },
        Nil =>
          if is_abs {
            stack               // ".." above root is root — drop
          } else {
            Cons(comp, stack)   // relative: preserve leading ".."
          },
      }
    } else {
      Cons(comp, stack)
    }
  }
}

fn __norm_walk(comps: List[String], is_abs: Bool, stack: List[String]) -> List[String] ![] {
  match comps {
    Nil => stack,
    Cons(comp, rest) => __norm_walk(rest, is_abs, __norm_push(stack, comp, is_abs)),
  }
}

// Join components with "/" (no leading/trailing slash).
fn __join_slash_acc(parts: List[String], acc: String) -> String ![] {
  match parts {
    Nil => acc,
    Cons(h, t) => __join_slash_acc(t, string_concat(string_concat(acc, "/"), h)),
  }
}
fn __join_slash(parts: List[String]) -> String ![] {
  match parts {
    Nil => "",
    Cons(h, t) => __join_slash_acc(t, h),
  }
}

// `path_normalize(p)` collapses ".", "..", and redundant slashes
// (posixpath.normpath). The empty string normalizes to ".". 2+ leading
// slashes collapse to one (our single deliberate divergence).
//   path_normalize("a/../b")="b"  path_normalize("a//b")="a/b"
//   path_normalize("")="."  path_normalize("/../a")="/a"  path_normalize("../a")="../a"
fn path_normalize(p: String) -> String ![] {
  if string_length(p) == 0 {
    "."
  } else {
    let is_abs: Bool = path_is_absolute(p);
    let stack: List[String] = __norm_walk(string_split(p, "/"), is_abs, Nil);
    let body: String = __join_slash(reverse(stack));
    if is_abs {
      string_concat("/", body)
    } else {
      if string_length(body) == 0 { "." } else { body }
    }
  }
}
```
